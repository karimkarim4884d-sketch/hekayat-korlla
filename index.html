async function fetchFromLink() { const status = document.getElementById('pTitle'); status.innerText = "جاري سحب القنوات.. انتظر ثواني ⏳"; try { // استخدام بروكسي "cors-anywhere" أو "allorigins" لكسر الحماية const proxyUrl = 'https://api.allorigins.win/raw?url='; const targetUrl = 'https://share.google/jm1ifPVaCW0x0QGjT'; const response = await fetch(proxyUrl + encodeURIComponent(targetUrl)); const data = await response.text(); // لو الداتا وصلت فاضية if (!data || data.trim() === "") { throw new Error("الملف فاضي أو الرابط مش شغال"); } const lines = data.split('\n'); let newCh = []; for (let i = 0; i < lines.length; i++) { if (lines[i].includes('#EXTINF')) { // استخراج الاسم بعد الفاصلة let name = lines[i].split(',')[1] || 'قناة بث'; // السطر اللي بعده هو الرابط let url = lines[i + 1]?.trim(); if (url && url.startsWith('http')) { newCh.push({ name: name.trim(), url: url }); } } } if (newCh.length > 0) { db = newCh; localStorage.setItem('zack_live_db', JSON.stringify(db)); render(); status.innerText = "مبروك! تم سحب " + db.length + " قناة ✅"; alert("تم تحديث الشبكة بـ " + db.length + " قناة"); } else { alert("لقيت الملف بس مش عارف أطلع منه روابط.. تأكد من صيغة الملف"); } } catch (error) { console.error(error); status.innerText = "فشل السحب.. تأكد من الإنترنت أو الرابط"; alert("الزرار معلق بسبب حماية المتصفح، جرب تفتحه من متصفح تاني أو حدث الصفحة"); } }
